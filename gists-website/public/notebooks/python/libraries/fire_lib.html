In [ ]:
import fire

def hello(name="World"):
  return "Hello %s!" % name

if __name__ == '__main__':
  fire.Fire(hello)

Then, from the command line, you can run:

python hello.py # Hello World!

python hello.py --name=David # Hello David!

python hello.py --help # Shows usage information.
